#include <stdio.h>
void addition(int A[20][20],int B[20][20],int r1,int c1){

int C[20][20];
 printf("addtion of matrix A and B:\n");
for(int i=0;i<r1;i++){
for(int j=0;j<c1;j++){
    C[i][j]=A[i][j]+B[i][j];
     printf("%d ",C[i][j]);
}
    printf("\n");
}

}
void multiplication(int A[20][20],int B[20][20],int r1,int c2,int c1){
int C[20][20];

 for(int i = 0; i < r1; i++) {
        for(int j = 0; j < c2; j++) {
            C[i][j] = 0;
            for(int k = 0; k < c1; k++) {
                C[i][j] += A[i][k] * B[k][j];
            }
            printf("%d ", C[i][j]);
        }
        printf("\n");
    }
}
void transpose(int A[20][20],int r1,int c1){
    printf("Transpose of matrix A:\n");
    for(int i=0;i<c1;i++){
        for(int j=0;j<r1;j++){
            printf("%d ",A[j][i]);
        }
        printf("\n");
    }
}
